# コンテナ監視スクリプト利用ガイド

## 概要
`container-monitor.sh` はPodmanコンテナの状態を監視し、問題が検出された場合にアラートを送信するスクリプトです。

## 機能
- コンテナの稼働状況監視
- リソース使用量監視 (CPU, メモリ)
- ヘルスチェック対応コンテナの状態確認
- Slack通知機能
- ログ記録機能

## 使用方法
```bash
# 通常実行
./container-monitor.sh

# 特定コンテナのみ監視
./container-monitor.sh --containers webapp,db

# デバッグモード
./container-monitor.sh --debug
```

## 設定ファイル
`.env` ファイルで以下を設定可能：
- `SLACK_WEBHOOK_URL`: Slack通知用Webhook URL
- `CONTAINER_LIST`: 監視対象コンテナリスト
- `CPU_THRESHOLD`: CPU使用率警告閾値 (%)
- `MEMORY_THRESHOLD`: メモリ使用率警告閾値 (%)

## 通知内容
- コンテナ停止検出時
- リソース使用率超過時
- ヘルスチェック失敗時
- システム異常時

# Podmanコンテナ管理のベストプラクティス

## 1. 監視スクリプトの活用
### 定期実行
- crontab等で定期的に実行し、常時監視を実現
- 推奨頻度: 5分間隔 (負荷と監視精度のバランス)

### アラート設定
- 閾値設定は環境に応じて調整
- Slack等での即時通知により迅速な対応が可能に

## 2. ヘルスチェックの実装
### コンテナ側
- `--health-cmd` を使って適切なヘルスチェックを実装
- ヘルスチェックは軽量で高速に完了するように設計

### 監視側
- ヘルスステータスに基づいた対応策を事前に準備
- 自動再起動などのオプションも検討

## 3. リソース管理
### 制限設定
- `--memory` および `--cpus` オプションで適切なリソース制限
- 過剰なリソース割当てを防ぎ、複数コンテナの共存を可能に

### モニタリング
- 定期的なリソース使用状況の記録
- ベンチマーキングによる適正リソース量の算出

## 4. ロギング
### 標準出力の活用
- コンテナの標準出力を適切に収集
- JSON形式などでの出力により解析性を向上

### 外部ログ管理
- ELK Stack等を用いたログ集約
- ログローテーションの設定によりディスク容量を節約

## 5. トラブルシューティング
### 一般的な問題
- コンテナの突然の停止
- リソース枯渇
- ネットワーク接続不良
- データ永続性の問題

### 対応手順
1. ログの確認 (`podman logs`)
2. コンテナ状態の確認 (`podman ps`, `podman inspect`)
3. リソース使用状況の確認 (`podman stats`)
4. 必要に応じて再起動 (`podman restart`)

===FILE_START:deliverables/container-helper-refined.sh===
#!/bin/bash

# Container Helper Script - Enhanced Version
# Purpose: Provides utility functions for managing Podman containers
# Features: Start, stop, restart, backup, restore, cleanup operations

set -e

# Load environment variables
ENV_FILE=".env"
if [[ -f "$ENV_FILE" ]]; then
    export $(cat $ENV_FILE | xargs)
fi

# Configuration
DEFAULT_CONTAINER_DIR="${CONTAINER_DATA_DIR:-./containers}"
BACKUP_DIR="${BACKUP_DIR:-./backups}"
LOG_FILE="${LOG_FILE:-./container-helper.log}"

# Logging function
log() {
    local level=$1
    shift
    local message="$*"
    local timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    echo "[$timestamp] [$level] $message" | tee -a "$LOG_FILE"
}

# Check if podman is installed
check_podman() {
    if ! command -v podman &> /dev/null; then
        log "ERROR" "Podman is not installed or not in PATH"
        exit 1
    fi
}

# Start container(s)
start_container() {
    local container_name=$1
    if [[ -z "$container_name" ]]; then
        log "ERROR" "Container name is required"
        return 1
    fi
    
    log "INFO" "Starting container: $container_name"
    
    if podman start "$container_name"; then
        log "SUCCESS" "Started container: $container_name"
        return 0
    else
        log "ERROR" "Failed to start container: $container_name"
        return 1
    fi
}

# Stop container(s)
stop_container() {
    local container_name=$1
    if [[ -z "$container_name" ]]; then
        log "ERROR" "Container name is required"
        return 1
    fi
    
    log "INFO" "Stopping container: $container_name"
    
    if podman stop "$container_name"; then
        log "SUCCESS" "Stopped container: $container_name"
        return 0
    else
        log "ERROR" "Failed to stop container: $container_name"
        return 1
    fi
}

# Restart container(s)
restart_container() {
    local container_name=$1
    if [[ -z "$container_name" ]]; then
        log "ERROR" "Container name is required"
        return 1
    fi
    
    log "INFO" "Restarting container: $container_name"
    
    if podman restart "$container_name"; then
        log "SUCCESS" "Restarted container: $container_name"
        return 0
    else
        log "ERROR" "Failed to restart container: $container_name"
        return 1
    fi
}

# Backup container
backup_container() {
    local container_name=$1
    if [[ -z "$container_name" ]]; then
        log "ERROR" "Container name is required"
        return 1
    fi
    
    local timestamp=$(date +%Y%m%d_%H%M%S)
    local backup_file="$BACKUP_DIR/${container_name}_backup_$timestamp.tar.gz"
    
    log "INFO" "Creating backup for container: $container_name"
    
    # Create backup directory if not exists
    mkdir -p "$BACKUP_DIR"
    
    if podman export "$container_name" | gzip > "$backup_file"; then
        log "SUCCESS" "Backup created: $backup_file"
        return 0
    else
        log "ERROR" "Failed to create backup for: $container_name"
        return 1
    fi
}

# Restore container from backup
restore_container() {
    local backup_file=$1
    local new_container_name=$2
    
    if [[ -z "$backup_file" ]] || [[ -z "$new_container_name" ]]; then
        log "ERROR" "Backup file and new container name are required"
        return 1
    fi
    
    if [[ ! -f "$backup_file" ]]; then
        log "ERROR" "Backup file does not exist: $backup_file"
        return 1
    fi
    
    log "INFO" "Restoring container from: $backup_file to: $new_container_name"
    
    if gunzip -c "$backup_file" | podman import - "$new_container_name"; then
        log "SUCCESS" "Container restored: $new_container_name"
        return 0
    else
        log "ERROR" "Failed to restore container from: $backup_file"
        return 1
    fi
}

# Clean stopped containers
cleanup_containers() {
    log "INFO" "Cleaning up stopped containers"
    
    local stopped_containers
    stopped_containers=$(podman ps -q -f status=exited)
    
    if [[ -z "$stopped_containers" ]]; then
        log "INFO" "No stopped containers to clean up"
        return 0
    fi
    
    if echo "$stopped_containers" | xargs podman rm; then
        log "SUCCESS" "Cleaned up stopped containers"
        return 0
    else
        log "ERROR" "Failed to clean up some stopped containers"
        return 1
    fi
}

# Clean unused images
cleanup_images() {
    log "INFO" "Cleaning up unused images"
    
    if podman image prune -f; then
        log "SUCCESS" "Cleaned up unused images"
        return 0
    else
        log "ERROR" "Failed to clean up unused images"
        return 1
    fi
}

# Show container statistics
show_stats() {
    local container_name=$1
    log "INFO" "Showing statistics"
    
    if [[ -n "$container_name" ]]; then
        podman stats --no-stream "$container_name"
    else
        podman stats --no-stream
    fi
}

# Main function
main() {
    local action=$1
    local target=$2
    local additional=$3
    
    check_podman
    
    case "$action" in
        "start")
            start_container "$target"
            ;;
        "stop")
            stop_container "$target"
            ;;
        "restart")
            restart_container "$target"
            ;;
        "backup")
            backup_container "$target"
            ;;
        "restore")
            restore_container "$target" "$additional"
            ;;
        "cleanup-containers")
            cleanup_containers
            ;;
        "cleanup-images")
            cleanup_images
            ;;
        "stats")
            show_stats "$target"
            ;;
        *)
            echo "Usage: $0 {start|stop|restart|backup|restore|cleanup-containers|cleanup-images|stats} [container_name] [additional_param]"
            echo "  start|stop|restart: [container_name]"
            echo "  backup: [container_name]"
            echo "  restore: [backup_file] [new_container_name]"
            echo "  stats: [container_name] (optional)"
            echo ""
            echo "Environment variables:"
            echo "  CONTAINER_DATA_DIR: Directory for container data (default: ./containers)"
            echo "  BACKUP_DIR: Directory for backups (default: ./backups)"
            echo "  LOG_FILE: Log file path (default: ./container-helper.log)"
            exit 1
            ;;
    esac
}

# Run main function with all arguments
main "$@"
